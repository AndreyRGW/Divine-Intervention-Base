### Transfer Variables to player heir ###
DI_transfer_variables = {
	save_scope_as = target

	if = {
		limit = {
			has_variable_list = DI_pinned_traits
		}
		every_in_list = {
			variable = DI_pinned_traits
			
			add_to_global_variable_list = {
				name = DI_pinned_traits
				target = this
			}
		}
	}
}
